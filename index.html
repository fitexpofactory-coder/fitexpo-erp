-- ════════════════════════════════════════════════════════════════ -- FitExpo ERP v4 — Stock Card (QR) tables -- รันใน Supabase → SQL Editor (รันซ้ำได้ปลอดภัย) -- ════════════════════════════════════════════════════════════════ -- 1) บัตรสต็อก: 1 แถว = 1 ล็อตของวัสดุ ที่ 1 โลเคชัน/พาเลท ----- create table if not exists public.stock_cards ( id uuid primary key default gen_random_uuid(), card_code text unique not null, -- รหัสใน QR เช่น SC-2026-000123 material_code text, -- → master_material.code material_name text, unit text, lot_no text, supplier text, coa_ref text, location text, pallet_no text, qty numeric default 0, -- ยอดคงเหลือปัจจุบัน qty_received numeric default 0, received_date date, expiry_date date, status text default 'active', -- active / hold / quarantine / depleted halal_ok boolean default true, remark text, created_by text, created_at timestamptz default now(), updated_at timestamptz default now() ); create index if not exists idx_sc_material on public.stock_cards(material_code); create index if not exists idx_sc_expiry on public.stock_cards(expiry_date); create index if not exists idx_sc_status on public.stock_cards(status); -- 2) ปูมการเคลื่อนไหว: รับเข้า/เบิก/ปรับ/ย้าย/นับ --------------- create table if not exists public.stock_card_moves ( id bigserial primary key, card_id uuid, card_code text, type text, -- in / out / adjust / move / count qty numeric, balance_after numeric, ref_doc text, -- WO / PO / GRN from_loc text, to_loc text, note text, by_user text, created_at timestamptz default now() ); create index if not exists idx_scm_card on public.stock_card_moves(card_code); -- 3) สิทธิ์เข้าถึงผ่าน anon key -------------------------------- alter table public.stock_cards enable row level security; alter table public.stock_card_moves enable row level security; drop policy if exists sc_all on public.stock_cards; drop policy if exists scm_all on public.stock_card_moves; create policy sc_all on public.stock_cards for all using (true) with check (true); create policy scm_all on public.stock_card_moves for all using (true) with check (true); grant all on public.stock_cards to anon, authenticated; grant all on public.stock_card_moves to anon, authenticated; grant usage, select on sequence public.stock_card_moves_id_seq to anon, authenticated; -- หมายเหตุ: รหัส SC-ปี-running ระบบออกให้อัตโนมัติฝั่งแอป -- และยอดจะ sync เข้า stock_material (qty_in/qty_out) ให้อัตโนมัติเมื่อรับ/เบิก/ปรับ
